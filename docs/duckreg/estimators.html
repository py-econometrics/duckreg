 39    def _parse_formula(self):
 40        lhs, rhs = self.formula.split("~")
 41        rhs_deparsed = rhs.split("|")
 42        covars, fevars = rhs.split("|") if len(rhs_deparsed) > 1 else (rhs, None)
 43
 44        self.outcome_vars = [x.strip() for x in lhs.split("+")]
 45        self.covars = [x.strip() for x in covars.split("+")]
 46        self.fevars = [x.strip() for x in fevars.split("+")] if fevars else []
 47        self.strata_cols = self.covars + self.fevars
 48
 49        if not self.outcome_vars:
 50            raise ValueError("No outcome variables found in the formula")

 56    def compress_data(self):
 57        agg_expressions = ["COUNT(*) as count"]
 58        agg_expressions += [f"SUM({var}) as sum_{var}" for var in self.outcome_vars]
 59        agg_expressions += [
 60            f"SUM(POW({var}, 2)) as sum_{var}_sq" for var in self.outcome_vars
 61        ]
 62        group_by_cols = ", ".join(self.strata_cols)
 63        self.agg_query = f"""
 64        SELECT {group_by_cols}, {', '.join(agg_expressions)}
 65        FROM {self.table_name}
 66        GROUP BY {group_by_cols}
 67        """
 68        self.df_compressed = pd.DataFrame(self.conn.execute(self.agg_query).fetchall())
 69        self.df_compressed.columns = (
 70            self.strata_cols
 71            + ["count"]
 72            + [f"sum_{var}" for var in self.outcome_vars]
 73            + [f"sum_{var}_sq" for var in self.outcome_vars]
 74        )
 75        create_means = "\n".join(
 76            [f"mean_{var} = sum_{var}/count" for var in self.outcome_vars]
 77        )
 78        self.df_compressed.eval(create_means, inplace=True)

 80    def collect_data(self, data: pd.DataFrame) -> pd.DataFrame:
 81        y = data.filter(
 82            regex=f"mean_{'(' +  '|'.join(self.outcome_vars) + ')'}", axis=1
 83        ).values
 84        X = data[self.covars].values
 85        n = data["count"].values
 86
 87        # y, X, w need to be two-dimensional for the demean function
 88        y = y.reshape(-1, 1) if y.ndim == 1 else y
 89        X = X.reshape(-1, 1) if X.ndim == 1 else X
 90
 91        if self.fevars:
 92            # fe needs to contain of only integers for
 93            # the demean function to work
 94            fe = _convert_to_int(data[self.fevars])
 95            fe = fe.reshape(-1, 1) if fe.ndim == 1 else fe
 96
 97            y, _ = demean(x=y, flist=fe, weights=n)
 98            X, _ = demean(x=X, flist=fe, weights=n)
 99        else:
100            X = np.c_[np.ones(X.shape[0]), X]
101
102        return y, X, n

109    def fit_vcov(self):
110        """compressed estimation of the heteroskedasticity-robust variance covariance matrix"""
111        self.se = "hc1"
112        y, X, n = self.collect_data(data=self.df_compressed)
113        betahat = wls(X, y, n).flatten()
114        # only works for single outcome for now
115        self.n_bootstraps = 0  # disable bootstrap
116        yprime = self.df_compressed[f"sum_{self.outcome_vars[0]}"].values.reshape(-1, 1)
117        yprimeprime = self.df_compressed[
118            f"sum_{self.outcome_vars[0]}_sq"
119        ].values.reshape(-1, 1)
120        yhat = (X @ betahat).reshape(-1, 1)
121        rss_g = (yhat**2) * n.reshape(-1, 1) - 2 * yhat * yprime + yprimeprime
122        bread = np.linalg.inv(X.T @ np.diag(n.flatten()) @ X)
123        meat = X.T @ np.diag(rss_g.flatten()) @ X
124        n_nk = n.sum() / (n.sum() - X.shape[1])
125        self.vcov = n_nk * (bread @ meat @ bread)

127    def estimate_feols(self):
128        if self.fevars:
129            fml = f"{'+'.join([f'mean_{x}' for x in self.outcome_vars])} ~ {' + '.join(self.covars)} | {' + '.join(self.fevars)}"
130        else:
131            fml = f"{'+'.join([f'mean_{x}' for x in self.outcome_vars])} ~ {' + '.join(self.covars)}"
132
133        fit = feols(
134            fml=fml,
135            data=self.df_compressed,
136            vcov="iid",  # most performant
137            weights="count",
138            weights_type="fweights",
139        )
140
141        return fit

143    def bootstrap(self):
144        self.se = "bootstrap"
145        if self.fevars:
146            boot_coefs = np.zeros(
147                (self.n_bootstraps, len(self.covars) * len(self.outcome_vars))
148            )
149        else:
150            boot_coefs = np.zeros(
151                (
152                    self.n_bootstraps,
153                    (len(self.strata_cols) + 1) * len(self.outcome_vars),
154                )
155            )
156
157        if not self.cluster_col:
158            # IID bootstrap
159            total_rows = self.conn.execute(
160                f"SELECT COUNT(DISTINCT {self.rowid_col}) FROM {self.table_name}"
161            ).fetchone()[0]
162            unique_rows = total_rows
163            self.bootstrap_query = f"""
164            SELECT {', '.join(self.strata_cols)}, {', '.join(["COUNT(*) as count"] + [f"SUM({var}) as sum_{var}" for var in self.outcome_vars])}
165            FROM {self.table_name}
166            GROUP BY {', '.join(self.strata_cols)}
167            """
168        else:
169            # Cluster bootstrap
170            unique_groups = self.conn.execute(
171                f"SELECT DISTINCT {self.cluster_col} FROM {self.table_name}"
172            ).fetchall()
173            unique_groups = [group[0] for group in unique_groups]
174            unique_rows = len(unique_groups)
175            self.bootstrap_query = f"""
176            SELECT {', '.join(self.strata_cols)}, {', '.join(["COUNT(*) as count"] + [f"SUM({var}) as sum_{var}" for var in self.outcome_vars])}
177            FROM {self.table_name}
178            WHERE {self.cluster_col} IN (SELECT unnest((?)))
179            GROUP BY {', '.join(self.strata_cols)}
180            """
181
182        for b in tqdm(range(self.n_bootstraps)):
183            resampled_rows = self.rng.choice(
184                unique_rows, size=unique_rows, replace=True
185            )
186            df_boot = pd.DataFrame(
187                self.conn.execute(
188                    self.bootstrap_query, [resampled_rows.tolist()]
189                ).fetchall()
190            )
191            df_boot.columns = (
192                self.strata_cols
193                + ["count"]
194                + [f"sum_{var}" for var in self.outcome_vars]
195            )
196            create_means = "\n".join(
197                [f"mean_{var} = sum_{var}/count" for var in self.outcome_vars]
198            )
199            df_boot.eval(create_means, inplace=True)
200
201            y, X, n = self.collect_data(data=df_boot)
202
203            boot_coefs[b, :] = wls(X, y, n).flatten()
204
205            # else np.diag() fails if input is not at least 1-dim
206            vcov = np.cov(boot_coefs.T)
207            vcov = np.expand_dims(vcov, axis=0) if vcov.ndim == 0 else vcov
208
209        return vcov

440    def prepare_data(self):
441        #   create_cohort_and_ever_treated_columns
442        self.temp_table_query = f"""

444            SELECT *, NULL::INTEGER AS cohort, NULL::INTEGER AS ever_treated
445            FROM {self.table_name};
446            UPDATE temp_{self.table_name} SET cohort = (
447                SELECT MIN({self.time_col})
448                FROM {self.table_name} AS p2
449                WHERE p2.{self.unit_col} = temp_{self.table_name}.{self.unit_col} AND
450                p2.{self.treatment_col} = 1
451            );
452            UPDATE temp_{self.table_name} SET cohort = NULL WHERE cohort = 2147483647;
453            UPDATE temp_{self.table_name} SET ever_treated = CASE WHEN cohort IS NOT NULL THEN 1 ELSE 0 END;
454        """
455        self.conn.execute(self.temp_table_query)

468                for i in range(self.num_periods + 1)
469            ]
470        )
471        # generate cohort dummies

473        for cohort in self.cohorts:

475                f"CASE WHEN cohort = {cohort} THEN 1 ELSE 0 END AS cohort_{cohort}"

488                )
489        self.treatment_dummies = ",\n".join(treatment_dummies)

494        SELECT
495            p.{self.unit_col},
496            p.{self.time_col},
497            p.{self.treatment_col},
498            p.{self.outcome_var},
499            -- Intercept (constant term)

501            -- cohort intercepts

503            -- Time dummies for each period
504            {self.time_dummies},
505            -- Treated group interacted with treatment time dummies
506            {self.treatment_dummies}

508            temp_{self.table_name} p;
509        """
